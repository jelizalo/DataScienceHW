# Create a high level snapshot (in table form) of the district's key metrics, including:

# Total Schools
total_schools = schools_csv_df['school'].count()

# Total Students
total_students = schools_csv_df['size'].sum()

# Total Budget
total_budget = schools_csv_df['budget'].sum()

# Average Math Score
average_math = students_csv_df['math_score'].mean()

# Average Reading Score
average_reading = students_csv_df['reading_score'].mean()

# % Passing Math
passing_math = students_csv_df.loc[students_csv_df['math_score']>=70]
passing_math_count = passing_math['math_score'].count()/total_students*100

# % Passing Reading
passing_reading = students_csv_df.loc[students_csv_df['reading_score']>=70]
passing_reading_count = passing_reading['reading_score'].count()/total_students*100

# Overall Passing Rate (Average of the above two)
overall_passing = (passing_math_count + passing_reading_count)/2

# Make a High Level Snapshot Dataframe
district_summary = pd.DataFrame({'Total Schools': [total_schools], 
                                 'Total Students':[total_students],
                                 'Total Budget': [total_budget],
                                'Average Math Score': [average_math],
                                'Average Reading Score': [average_reading],
                                '% Passing Math': [passing_math_count],
                                '% Passing Reading': [passing_reading_count],
                                'Overall % Passing Rate': [overall_passing]})

school_summary = school_summary.drop(["School ID",
                                      "('mean', 'Passed Math')", 
                                      "('mean', 'Passed Reading')",
                                      "('sum', 'math_score')",
                                      "('sum', 'reading_score')"], axis=1)

# School Name
school_summary = school_summary.rename(columns = {'school':'School Name'})

# School Type
school_summary = school_summary.rename(columns = {'type':'School Type'})

# Total Students
school_summary = school_summary.rename(columns = {'size':'Total Students'})

# Total School Budget
school_summary = school_summary.rename(columns = {'budget':'Total School Budget'})

#student_budget
school_summary['Student Budget'] = school_summary['Total School Budget'] / school_summary['Total Students']

# Average Math Score
school_summary = school_summary.rename(columns = {"('mean', 'math_score')":'Average Math Score'})

# Average Reading Score
school_summary = school_summary.rename(columns = {"('mean', 'reading_score')":'Average Reading Score'})

# % Passing Math
school_summary['% Passing Math'] = school_summary["('sum', 'Passed Math')"] / school_summary['Total Students']*100

# % Passing Reading
school_summary['% Passing Reading'] = school_summary["('sum', 'Passed Reading')"] / school_summary['Total Students']*100

# Overall Passing Rate (Average of the above two)
school_summary['Overall % Passing Rate'] = (school_summary['% Passing Math'] + school_summary['% Passing Reading'])/2

# Clean up Data
school_summary = school_summary.drop(["('sum', 'Passed Math')",
                                      "('sum', 'Passed Reading')"], axis=1)
